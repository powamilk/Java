var slideIndex = 0; var slides = document.getElementsByClassName("slide"); var intervalId; function showSlide(n) { for (var i = 0; i < slides.length; i++) { slides[i].style.display = "none"; } slides[n].style.display = "block"; } function nextSlide() { slideIndex++; if (slideIndex >= slides.length) { slideIndex = 0; } showSlide(slideIndex); } function prevSlide() { slideIndex--; if (slideIndex < 0) { slideIndex = slides.length - 1; } showSlide(slideIndex); } function startSlideshow() { intervalId = setInterval(nextSlide, 1000 ); } function stopSlideshow() { clearInterval(intervalId); } document.getElementById("prev").addEventListener("click", prevSlide); document.getElementById("stop").addEventListener("click", stopSlideshow); document.getElementById("start").addEventListener("click", startSlideshow); document.getElementById("next").addEventListener("click", nextSlide); startSlideshow();
